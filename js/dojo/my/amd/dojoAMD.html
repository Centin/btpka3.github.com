最佳实践？—— 定义一个组件
xxx/myModule.js
--------------------- 定义可重复使用的组件。回调函数只是定义，而不执行。
define([
    "exports",
    "dojo/dom",
    "dojo/parser",
    ...
],
function(exports, dom, parser, ...) {
  var unExportedVar = ...;
  function unExportedFunction(){ ... }

  // --------------Exported properties
  exports.exportedVar = ...;
  exports.exportedFunction = ...;
});

yyy/myCode.js
--------------------- 使用--一般来说不具备可重复使用。回调函数会立即执行。
require([
    "dojo/dom",
    "xxx/myModule",
    ...
],
function(dom, myModule, ...) {
  myModule.exportedFunction(...)
  ...
});
